    <script>
        let entries = [];

        function toggleForm() {
            const formContainer = document.getElementById('formContainer');
            formContainer.classList.toggle('active');

            if (!formContainer.classList.contains('active')) {
                clearForm();
            }
        }

        function clearForm() {
            document.getElementById('typeSelect').value = 'feat';
            document.getElementById('scopeInput').value = '';
            document.getElementById('labelsInput').value = '';
            document.getElementById('messageInput').value = '';
            document.getElementById('breakingCheck').checked = false;
        }

        function getNextVersion() {
            if (entries.length === 0) return '1.0.0';

            const lastVersion = entries[0].version || '1.0.0';
            const [major, minor, patch] = lastVersion.split('.').map(Number);

            const type = document.getElementById('typeSelect').value;
            const breaking = document.getElementById('breakingCheck').checked;

            if (breaking) return `${major + 1}.0.0`;
            if (type === 'feat') return `${major}.${minor + 1}.0`;
            return `${major}.${minor}.${patch + 1}`;
        }

        function getTypeIcon(type) {
            const icons = {
                feat: '‚ú®',
                fix: 'üêõ',
                docs: 'üìù',
                style: 'üíÑ',
                refactor: '‚ôªÔ∏è',
                perf: '‚ö°',
                test: '‚úÖ',
                chore: 'üîß'
            };
            return icons[type] || 'üîß';
        }

        function addEntry() {
            const type = document.getElementById('typeSelect').value;
            const scope = document.getElementById('scopeInput').value;
            const labelsInput = document.getElementById('labelsInput').value;
            const message = document.getElementById('messageInput').value;
            const breaking = document.getElementById('breakingCheck').checked;

            if (!message.trim()) {
                alert('Por favor ingresa un mensaje');
                return;
            }

            // Mapear tipos a labels de GitHub
            const typeToLabels = {
                feat: ['enhancement', 'Semver-Minor'],
                fix: ['bug', 'Semver-Patch'],
                docs: ['docs'],
                style: ['style'],
                refactor: ['refactor'],
                perf: ['performance', 'Semver-Patch'],
                test: ['test'],
                chore: ['chore']
            };

            let labels = typeToLabels[type] || [];
            if (breaking) {
                labels.push('breaking-change', 'Semver-Major');
            }
            if (labelsInput.trim()) {
                labels = labels.concat(labelsInput.split(',').map(l => l.trim()));
            }

            const newEntry = {
                id: Date.now(),
                type: type,
                scope: scope,
                labels: labels,
                message: message,
                breaking: breaking,
                date: new Date().toISOString(),
                version: getNextVersion()
            };

            entries.unshift(newEntry);
            saveToFile();
            renderEntries();
            toggleForm();
        }

        function saveToFile() {
            const dataStr = JSON.stringify(entries, null, 2);
            const dataBlob = new Blob([dataStr], { type: 'application/json' });
            const url = URL.createObjectURL(dataBlob);
            const link = document.createElement('a');
            link.href = url;
            link.download = 'my-changelog.json';
            link.click();
            URL.revokeObjectURL(url);
        }

        function renderEntries() {
            var container = document.getElementById('entriesContainer');

            if (entries.length === 0) {
                container.innerHTML = '<div class="empty-state">' +
                    '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2">' +
                    '<path d="M14 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V8z"/>' +
                    '<polyline points="14 2 14 8 20 8"/>' +
                    '<line x1="16" y1="13" x2="8" y2="13"/>' +
                    '<line x1="16" y1="17" x2="8" y2="17"/>' +
                    '<polyline points="10 9 9 9 8 9"/>' +
                    '</svg>' +
                    '<p>No hay entradas en el changelog a√∫n</p>' +
                    '<small>Agrega tu primera entrada usando el bot√≥n de arriba</small>' +
                    '</div>';
                return;
            }

            var html = '';
            for (var i = 0; i < entries.length; i++) {
                var entry = entries[i];
                var date = new Date(entry.date);
                var formattedDate = date.toLocaleDateString('es-MX', {
                    year: 'numeric',
                    month: 'long',
                    day: 'numeric'
                });

                var scopeBadge = entry.scope ? '<span class="badge badge-scope">' + entry.scope + '</span>' : '';
                var breakingBadge = entry.breaking ? '<span class="badge badge-breaking">‚ö†Ô∏è BREAKING</span>' : '';
                var labelsBadges = (entry.labels || []).map(label => '<span class="badge badge-label">' + label + '</span>').join('');

                html += '<div class="entry-card">' +
                    '<div class="entry-badges">' +
                    '<span class="badge badge-' + entry.type + '">' +
                    getTypeIcon(entry.type) + ' ' + entry.type +
                    '</span>' +
                    scopeBadge +
                    breakingBadge +
                    labelsBadges +
                    '</div>' +
                    '<p class="entry-message">' + entry.message + '</p>' +
                    '<div class="entry-meta">' +
                    '<span>üè∑Ô∏è v' + entry.version + '</span>' +
                    '<span>üìÖ ' + formattedDate + '</span>' +
                    '</div>' +
                    '</div>';
            }
            container.innerHTML = html;
        }

        function exportMarkdown() {
            if (entries.length === 0) {
                alert('No hay entradas para exportar');
                return;
            }

            let markdown = '# Changelog\n\n';

            const groupedByVersion = entries.reduce((acc, entry) => {
                if (!acc[entry.version]) acc[entry.version] = [];
                acc[entry.version].push(entry);
                return acc;
            }, {});

            Object.keys(groupedByVersion).forEach(version => {
                const versionEntries = groupedByVersion[version];
                const date = new Date(versionEntries[0].date);
                markdown += `## [${version}] - ${date.toLocaleDateString()}\n\n`;

                const types = ['feat', 'fix', 'docs', 'style', 'refactor', 'perf', 'test', 'chore'];
                types.forEach(type => {
                    const typeEntries = versionEntries.filter(e => e.type === type);
                    if (typeEntries.length > 0) {
                        markdown += `### ${type.charAt(0).toUpperCase() + type.slice(1)}\n\n`;
                        typeEntries.forEach(entry => {
                            const scope = entry.scope ? `**${entry.scope}**: ` : '';
                            const breaking = entry.breaking ? ' ‚ö†Ô∏è BREAKING CHANGE' : '';
                            markdown += `- ${scope}${entry.message}${breaking}\n`;
                        });
                        markdown += '\n';
                    }
                });
            });

            const blob = new Blob([markdown], { type: 'text/markdown' });
            const url = URL.createObjectURL(blob);
            const link = document.createElement('a');
            link.href = url;
            link.download = 'CHANGELOG.md';
            link.click();
            URL.revokeObjectURL(url);
        }

        // Validaci√≥n del bot√≥n
        document.getElementById('messageInput').addEventListener('input', function () {
            const addBtn = document.getElementById('addBtn');
            addBtn.disabled = !this.value.trim();
        });

        // Inicializar
        renderEntries();
    </script>
